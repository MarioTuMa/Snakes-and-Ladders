<script type="text/javascript">
  
var score=0; 
var ladders = [4, 9,20,28,40,51,63,71];
var snakes  = [17,62,87,54,64,93,95,99];
var ladderend = [14,31,38,84,59,67,81,91];
var snaketail = [7,19,24,34,60,73,75,78];

function wait(){
  score=0;
document.getElementById("notifier").innerHTML="Your at " +score;
  
}

$("button#restart").click(function(){
wait();
  
});

function check(x){
  document.getElementById("notifier").innerHTML="You're at " + x;
  for (i=0;i<8;i++){
    
    if ( x == ladders [i] ) {
    var laddernewscore=ladderend[i]; 
    console.log(laddernewscore+"ladder");  
     document.getElementById("notifier").innerHTML="You landed on "+score+" but fortunately, a ladder took you up to " + laddernewscore;
      score=laddernewscore;
    }  
  else{ if(x==snakes[i]){
      var snakenewscore=snaketail[i];
    console.log(snakenewscore+"snake");
    document.getElementById("notifier").innerHTML="You landed on "+score+" but unfortunately, a snake took you down to " + snakenewscore;
      score=snakenewscore;

  }
    }
        }
    
    
  
  if (x>=100){

    var newscore=100-(x-100);
    console.log(newscore);
    document.getElementById("notifier").innerHTML="You went too high and rebounded to " +newscore;
          score=newscore;
    if (x==100){
       document.getElementById("notifier").innerHTML="You win";
      setTimeout(wait, 3000);
      
      score=newscore;
      
    }
  }  
  
}
     
  

$("button#roller").click(function(){
    
  var sides = $("input#sides").val();
    if (sides<=100){
    var rollcount = Math.ceil(sides*(Math.random()));
    score = rollcount+score;
    console.log(rollcount);  
    console.log(score);
      
    check(score);  
    }
  
  else{
    alert("make the number of sides smaller");
  }
  });
